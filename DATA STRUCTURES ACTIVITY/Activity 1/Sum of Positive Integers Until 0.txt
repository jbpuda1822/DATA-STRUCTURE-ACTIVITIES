//Sum of Positive Integers Until 0
#include <iostream>
using namespace std;

int main() {
    int num;
    int sum = 0;

    cout << "Enter integers: " << endl;
    cin >> num;

    while (num != 0) {
        if (num > 0) {
            sum += num;
        }
        cin >> num;
    }
    cout << "The sum of all entered positive integers is: " << sum << endl;
    
    return 0;
}
